```php
//---------------Views & Controllers

//---------------Create & Display views

Route::get('/jobs', function () {
    return view('jobs');
})->name('jobs');

Route::get('/jobs/profile', function () {
    return view('jobs.index');
});

Route::get('/jobs/create', function () {
    return view('jobs.create');
})->name('jobs.create');

// ---------------jobs.php or views/jobs/index.php
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Job listings</title>
</head>
<body>
    <h1>Available Jobs</h1>
    <ul>
        <li>Web Developer</li>
        <li>Database Admin</li>
        <li>Software Engineer</li>
        <li>Systems Analyst</li>
    </ul>
</body>
</html>

// ----------------Passing Data to Views
//First Way

Route::get('/jobs/profile', function () {
    return view('jobs.index', [
        'title' => 'Available Jobs',
    ]);
});


//Second Way

Route::get('/jobs/profile', function () {
    return view('jobs.index')->with('title', 'Available Jobs');
});

//third Way

Route::get('/jobs/profile', function () {
    $title = 'Available Jobs';
    return view('jobs.index', compact('title'));
});

Route::get('/jobs/profile1', function () {
    $title = 'Available Jobs';
    $jobs = [
        'Web Developer',
        'Database Admin',
        'Software Engineer',
        'Systems Analyst',
    ];
    return view('jobs.index1', compact('title', 'jobs'));
});

//index1.php
<body>
    <h1><?php echo $title; ?></h1>
    <ul>
        <li>Web Developer</li>
        <li>Database Admin</li>
        <li>Software Engineer</li>
        <li>Systems Analyst</li>
    </ul>
</body>

//index1.php
<body>
    <h1><?php echo $title; ?></h1>
    <ul>
        <?php foreach ($jobs as $job): ?>
            <li><?php echo $job; ?> </li>
        <!--or-->
        <!--<li><?php echo htmlspecialchars($job, ENT_QUOTES) ?> use for security </li>-->
        <?php endforeach;?>
    </ul>
</body>

```
